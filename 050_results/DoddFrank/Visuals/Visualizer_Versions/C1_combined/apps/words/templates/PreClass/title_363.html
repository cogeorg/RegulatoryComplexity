class = "ex8">(i) a collateralized mortgage obligation;
class = "ex8">(ii) a collateralized debt obligation;
class = "ex8">(iii) a collateralized bond obligation;
class = "ex8">(iv) a collateralized debt obligation of asset-backed securities;
class = "ex8">(v) a collateralized debt obligation of collateralized debt obligations; and
class = "ex8">(vi) a security that the Commission, by rule, determines to be an asset-backed security for purposes of this section; and
class = "ex7">(B) does not include a security issued by a finance subsidiary held by the parent company or a company controlled by the parent company, if none of the securities issued by the finance subsidiary are held by an entity that is not controlled by the parent company.
